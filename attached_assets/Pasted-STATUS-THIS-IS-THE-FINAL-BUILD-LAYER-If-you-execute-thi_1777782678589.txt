app.post("/api/submit-score", async (req, res) => {
  const { userId, score, blendName, locationId } = req.body;

  await db.insert("leaderboard_live", {
    user_id: userId,
    score,
    blend_name: blendName,
    location_id: locationId,
    timestamp: new Date()
  });

  res.sendStatus(200);
});
🔥 GET LEADERBOARD
app.get("/api/leaderboard/:locationId", async (req, res) => {
  const data = await db.query(`
    SELECT * FROM leaderboard_live
    WHERE location_id = $1
    ORDER BY score DESC
    LIMIT 5
  `, [req.params.locationId]);

  res.json(data.rows);
});
