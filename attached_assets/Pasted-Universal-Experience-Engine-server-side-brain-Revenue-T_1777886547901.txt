🧠 EXPERIENCE ENGINE (CORE LOGIC)

server/engine/experienceEngine.ts

export function calculateScore(choices: any) {
  let score = 0;

  if (choices.vibe === "social") score += 20;
  if (choices.strength === "heavy") score += 30;
  if (choices.flavor === "smoky") score += 40;

  return score;
}

export function generateProfile(score: number) {
  if (score > 80) return "Bold Dominant";
  if (score > 50) return "Balanced Explorer";
  return "Smooth Casual";
}
💰 REVENUE ENGINE

server/engine/revenueEngine.ts

export function generateReward(score: number) {
  const rand = Math.random();

  if (score > 80 && rand > 0.7) {
    return { type: "premium", value: "Free Upgrade" };
  }

  if (score > 50) {
    return { type: "discount", value: "10% Off" };
  }

  return { type: "small", value: "5% Off" };
}

export async function updateVenueStats(venueId: string, score: number) {
  await db.execute(`
    UPDATE venue_stats
    SET total_sessions = total_sessions + 1,
        avg_score = (avg_score + ${score}) / 2
    WHERE venue_id = ${venueId}
  `);
}
🌐 EXPERIENCE ROUTES

server/routes/experienceRoutes.ts

import express from "express";
import { calculateScore, generateProfile } from "../engine/experienceEngine";
import { generateReward } from "../engine/revenueEngine";
import { updateLeaderboard, updateVenueStats } from "../engine/competitionEngine";
import { db } from "../db";
import { experienceSessions } from "../db/schema";

const router = express.Router();

router.post("/complete", async (req, res) => {
  const { sessionId, venueId, mode, choices } = req.body;

  const score = calculateScore(choices);
  const profile = generateProfile(score);
  const reward = generateReward(score);

  await db.insert(experienceSessions).values({
    sessionId,
    venueId,
    mode,
    score,
    reward: reward.value,
    profile,
    completed: true
  });

  await updateLeaderboard(sessionId, venueId, score);
  await updateVenueStats(venueId, score);

  res.json({
    score,
    profile,
    reward
  });
});

export default router;
