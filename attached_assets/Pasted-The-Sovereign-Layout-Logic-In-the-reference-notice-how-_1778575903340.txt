/* Add this logic to the center tiles */
const StatusRing = ({ value, label }) => (
  <div className="relative w-20 h-20 flex items-center justify-center">
    <svg className="w-full h-full transform -rotate-90">
      <circle cx="40" cy="40" r="36" stroke="rgba(255,255,255,0.1)" strokeWidth="4" fill="transparent" />
      <motion.circle 
        cx="40" cy="40" r="36" stroke="#00d4ff" strokeWidth="4" fill="transparent"
        strokeDasharray="226"
        initial={{ strokeDashoffset: 226 }}
        animate={{ strokeDashoffset: 226 - (226 * value) / 100 }}
        style={{ filter: 'drop-shadow(0 0 5px #00d4ff)' }}
      />
    </svg>
    <div className="absolute text-center">
      <span className="block text-white font-bold text-lg">{value}%</span>
      <span className="block text-[8px] text-white/40 uppercase">{label}</span>
    </div>
  </div>
