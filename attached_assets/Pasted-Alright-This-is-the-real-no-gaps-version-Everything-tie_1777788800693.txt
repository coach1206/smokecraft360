export function checkReorder(inventory: any) {
  return Object.entries(inventory)
    .filter(([_, item]: any) => item.stock < 5)
    .map(([name]) => name);
}

🧠 ENGINE

server/services/menu/layoutEngine.ts

export function optimizeMenuLayout(items: any[]) {
  return items
    .map(item => {
      let score = 0;

      score += item.margin * 2;
      score += item.popularity;
      score += item.conversionRate * 3;

      if (!item.available) score -= 100;

      return { ...item, score };
    })
    .sort((a, b) => b.score - a.score);
}
🔌 ROUTE
router.post("/optimize-layout", (req, res) => {
  const optimized = optimizeMenuLayout(req.body.items);
  res.json(optimized);
});

    margin: ((item.price - item.cost) / item.price) * 100,

}
