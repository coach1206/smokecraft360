This replaces your irrelevant page. It uses framer-motion for the "Tinder" feel and talks to the EmberHeartbeat.

TypeScript
import { motion, useMotionValue, useTransform } from 'framer-motion';
import { useEFE } from '../hooks/useEFE';

const cards = [
  { id: 'earthy', label: 'EARTHY', color: '#4A5D23' },
  { id: 'spiced', label: 'SPICED', color: '#D35400' },
  { id: 'cedar',  label: 'CEDAR',  color: '#A0522D' },
  { id: 'leather',label: 'LEATHER',color: '#4B3621' }
];

export const SwipeRitual = () => {
  const { advance } = useEFE();
  const x = useMotionValue(0);
  
  // Bridge UI to the EmberHeartbeat scale
  const emberScale = useTransform(x, [-200, 0, 200], [0.5, 1, 2]);

  const handleSwipe = (profile) => {
    // Send to Infrastructure Bridge
    fetch('/api/telemetry/ritual', { method: 'POST', body: JSON.stringify({ profile }) });
    // Advance logic
    if (isLastCard) advance();
  };

  return (
    <div className="ritual-container">
      <motion.div style={{ scale: emberScale }}>
        <EmberHeartbeat />
      </motion.div>
      
      {cards.map(card => (
        <motion.div 
          drag="x"
          onDragEnd={(e, info) => info.offset.x > 100 && handleSwipe(card.id)}
          className="sensory-card"
        >
          {card.label}
        </motion.div>
      ))}
    </div>
  );
};
