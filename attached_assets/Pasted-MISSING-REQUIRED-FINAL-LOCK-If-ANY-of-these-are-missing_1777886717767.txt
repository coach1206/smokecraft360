import Stripe from "stripe";
const stripe = new Stripe(process.env.STRIPE_SECRET!);

export async function createVenueSubscription(venueId: string, priceId: string) {
  const customer = await stripe.customers.create({
    metadata: { venueId }
  });

  const sub = await stripe.subscriptions.create({
    customer: customer.id,
    items: [{ price: priceId }]
  });

  return sub;
}

export async function processPayout(venueId: string, amount: number, stripeAccountId: string) {
  const transfer = await stripe.transfers.create({
    amount,
    currency: "usd",
    destination: stripeAccountId
  });

  await db.insert(venuePayouts).values({
    venueId,
    amount,
    status: "paid",
    stripeTransferId: transfer.id
  });

  return transfer;
}

export async function optimizeRewards(venueId: string) {
  const data = await db.execute(`
    SELECT AVG(amount) as avgRevenue FROM revenue_log WHERE venue_id = '${venueId}'

  const avg = data[0].avgrevenue;

  if (avg < 1000) {
