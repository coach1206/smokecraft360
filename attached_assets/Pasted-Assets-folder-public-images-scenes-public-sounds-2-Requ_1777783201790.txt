📁 FILE 2 — SCENE + SOUND ENGINE
import { useEffect, useRef, useState } from "react";

export default function SceneEngine({ scene }: any) {
  const videoRef = useRef(null);
  const audioRef = useRef<HTMLAudioElement | null>(null);

  const [current, setCurrent] = useState(scene);

  useEffect(() => {
    setCurrent(scene);

    // SOUND CONTROL
    if (audioRef.current) {
      audioRef.current.pause();
    }

    const audio = new Audio(scene.sound);
    audio.loop = true;
    audio.volume = 0.4;

    audio.play().catch(() => {});
    audioRef.current = audio;

  }, [scene]);

  return (
    <div className="absolute inset-0 z-0">
      
      <video
        ref={videoRef}
        autoPlay
        muted
        loop
        playsInline
        className="w-full h-full object-cover transition-opacity duration-700"
      >
        <source src={current.video} type="video/mp4" />
      </video>

      <div className="absolute inset-0 bg-black/60 backdrop-blur-sm" />

    </div>
  );
}

app.post("/api/ai/mood", async (req, res) => {
  const profile = req.body;

  const mood = `
User prefers ${profile.flavors}.
Tone: ${profile.personality}

Respond with lounge-style energy.
`;

  const response = await openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{ role: "user", content: mood }]
  });

  res.json({ text: response.choices[0].message.content });
});
