const AmbiContext = createContext(null);

export const AmbiProvider = ({ children }) => {
  const [currentAtmosphere, setCurrentAtmosphere] = useState({
    lightingLux: 100,
    lightingColorTemp: '4000K',
    audioSoundscape: 'Standard Ambient Floor',
    aromaSaturation: 0,
    activeZoneId: 'VIP_Lounge_Main'
  });

  const triggerAmbiScene = (moduleName, variantName) => {
    console.log(`[TACTILE SENSORY] Micro-haptic triggered for ${moduleName}`);
    let targetProfile = {};
    
    if (moduleName === 'WineCraft') {
      targetProfile = {
        lightingLux: 30,
        lightingColorTemp: '2400K (Velvet Burgundy Amber)',
        audioSoundscape: 'Boutique Sommelier Jazz Dispersion Floor',
        aromaSaturation: 50,
      };
    }

    setCurrentAtmosphere(prev => ({ ...prev, ...targetProfile }));
  };

  return (
    <AmbiContext.Provider value={{ currentAtmosphere, triggerAmbiScene }}>
      {children}
    </AmbiContext.Provider>
  );
};

// ==========================================
// HIGH SECURITY VAULT COMPONENT (PIN: 3600)
// ==========================================
export const SovereignVaultGate = ({ children }) => {
  const [pin, setPin] = useState('');
  const [isUnlocked, setIsUnlocked] = useState(false);

  const handlePress = (digit) => {
    const newPin = pin + digit;
    if (newPin.length <= 4) setPin(newPin);
    if (newPin.length === 4) {
      if (newPin === '3600') {
        setIsUnlocked(true);
        setPin('');
      } else {
        Alert.alert("SECURITY ERROR", "Clearance PIN Rejected.");
        setPin('');
      }
    }
  };

  if (isUnlocked) {
    return (
      <View style={styles.vaultContainer}>
        <View style={styles.vaultHeader}>
          <Text style={styles.vaultTitle}>⚡ SOVEREIGN SECURE TERMINAL ACTIVE</Text>
          <TouchableOpacity style={styles.lockBtn} onPress={() => setIsUnlocked(false)}>
            <Text style={styles.lockBtnText}>LOCK</Text>
          </TouchableOpacity>
        </View>
        {children}
      </View>
    );
  }

  return (
    <View style={styles.gateContainer}>
      <Text style={styles.gateLabel}>SOVEREIGN SECURITY CLEARANCE</Text>
      <Text style={styles.pinIndicator}>{"* ".repeat(pin.length) + "_ ".repeat(4 - pin.length)}</Text>
      <View style={styles.keypad}>
        {['1', '2', '3', '4', '5', '6', '7', '8', '9', '0'].map((d) => (
          <TouchableOpacity key={d} style={styles.key} onPress={() => handlePress(d)}>
            <Text style={styles.keyText}>{d}</Text>
          </TouchableOpacity>
        ))}
        <TouchableOpacity style={styles.clearKey} onPress={() => setPin('')}>
          <Text style={styles.keyText}>RESET</Text>
        </TouchableOpacity>
      </View>
    </View>
  );
};
