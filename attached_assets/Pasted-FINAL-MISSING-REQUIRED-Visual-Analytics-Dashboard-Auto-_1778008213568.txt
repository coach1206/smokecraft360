import { db } from "../db";import { sql } from "drizzle-orm";export async function getVenueAnalytics(venueId: string) {  const result = await db.execute(sql`    SELECT      COUNT(*) FILTER (WHERE event_type = 'SESSION_START') AS total_sessions,      COUNT(*) FILTER (WHERE event_type = 'PRODUCT_VIEWED') AS product_views,      COUNT(*) FILTER (WHERE event_type = 'PRODUCT_SELECTED') AS product_selections,      COUNT(*) FILTER (WHERE event_type = 'PRODUCT_PURCHASED') AS purchases,      COUNT(*) FILTER (WHERE event_type = 'UPSELL_ACCEPTED') AS upsells,      COUNT(*) FILTER (WHERE event_type = 'LOYALTY_USED') AS loyalty_used,      COALESCE(AVG(price) FILTER (WHERE event_type = 'PRODUCT_PURCHASED'), 0) AS average_spend    FROM event_logs    WHERE venue_id = ${venueId};  `);  return result.rows?.[0] || {};}export async function getTopProducts(venueId: string) {  const result = await db.execute(sql`    SELECT product_id, COUNT(*) AS total    FROM event_logs    WHERE venue_id = ${venueId}      AND event_type = 'PRODUCT_PURCHASED'      AND product_id IS NOT NULL    GROUP BY product_id    ORDER BY total DESC    LIMIT 10;  `);  return result.rows || [];}export async function getFlavorTrends(venueId: string) {  const result = await db.execute(sql`    SELECT metadata->>'flavor' AS flavor, COUNT(*) AS total    FROM event_logs    WHERE venue_id = ${venueId}      AND metadata->>'flavor' IS NOT NULL    GROUP BY flavor    ORDER BY total DESC    LIMIT 10;  `);  return result.rows || [];}export async function getStrengthTrends(venueId: string) {  const result = await db.execute(sql`    SELECT metadata->>'strength' AS strength, COUNT(*) AS total    FROM event_logs    WHERE venue_id = ${venueId}      AND metadata->>'strength' IS NOT NULL    GROUP BY strength    ORDER BY total DESC    LIMIT 10;  `);  return result.rows || [];}

import {  getVenueAnalytics,  getFlavorTrends,  getStrengthTrends,  getTopProducts,} from "./analyticsAggregationService";export async function generateVenueInsights(venueId: string) {  const analytics: any = await getVenueAnalytics(venueId);  const flavors: any[] = await getFlavorTrends(venueId);  const strengths: any[] = await getStrengthTrends(venueId);  const topProducts: any[] = await getTopProducts(venueId);  const productViews = Number(analytics.product_views || 0);  const purchases = Number(analytics.purchases || 0);  const upsells = Number(analytics.upsells || 0);  const selections = Number(analytics.product_selections || 0);  const conversionRate = productViews > 0 ? (purchases / productViews) * 100 : 0;  const upsellRate = selections > 0 ? (upsells / selections) * 100 : 0;  const insights = [];  if (flavors[0]) {    insights.push({      type: "FLAVOR_TREND",      title: "Top Flavor Trend",      summary: `${flavors[0].flavor} is currently the strongest flavor preference for this venue.`,      severity: "INFO",    });  }  if (strengths[0]) {    insights.push({      type: "STRENGTH_TREND",      title: "Top Strength Trend",      summary: `${strengths[0].strength} strength products are getting the most engagement.`,      severity: "INFO",    });  }  if (conversionRate < 20 && productViews > 20) {    insights.push({      type: "CONVERSION_WARNING",      title: "High Interest, Low Purchases",      summary:        "Customers are viewing products but not buying at the same rate. This may point to price friction, weak product placement, or poor pairing confidence.",      severity: "WARNING",    });  }  if (upsellRate < 10 && selections > 20) {    insights.push({      type: "UPSELL_OPPORTUNITY",      title: "Upsell Opportunity",      summary:        "Customers are selecting products, but upsells are not converting strongly. Add stronger pairing bundles or staff prompts.",      severity: "OPPORTUNITY",    });  }  if (topProducts[0]) {    insights.push({      type: "TOP_PRODUCT",      title: "Top Product Movement",      summary: `${topProducts[0].product_id} is leading purchase activity. Consider featuring it inside the experience flow.`,      severity: "INFO",    });  }  return {    venueId,    conversionRate,    upsellRate,    insights,  };}
