SwipeCard (Framer Motion)
import { motion } from "framer-motion";

export default function SwipeCard({ data, onSwipe }) {
  return (
    <motion.div
      drag
      onDragEnd={(e, info) => {
        if (info.offset.x > 100) onSwipe("right", data);
        else if (info.offset.x < -100) onSwipe("left", data);
        else if (info.offset.y < -100) onSwipe("up", data);
        else if (info.offset.y > 100) onSwipe("down", data);
      }}
      style={{
        width: 320, height: 420, borderRadius: 16,
        background: "#111", color: "#fff",
        display: "flex", alignItems: "center", justifyContent: "center",
        position: "absolute"
      }}
    >
      <h2>{data.title}</h2>
    </motion.div>
  );
}
Voice (queued)
const queue = [];
let speaking = false;

export function speak(text) {
  queue.push(text);
  if (!speaking) next();
}

function next() {
  if (!queue.length) { speaking = false; return; }
  speaking = true;
  const u = new SpeechSynthesisUtterance(queue.shift());
  u.onend = next;
  speechSynthesis.speak(u);
}
