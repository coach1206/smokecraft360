import { WebSocketServer, WebSocket } from 'ws';

export function initializeSyncEngine(server: any) {
  const wss = new WebSocketServer({ noServer: true });
  const activeTerminals = new Set<WebSocket>();

  server.on('upgrade', (request: any, socket: any, head: any) => {
    if (request.url === '/ws/theme-sync') {
      wss.handleUpgrade(request, socket, head, (ws) => {
        wss.emit('connection', ws, request);
      });
    }
  });

  wss.on('connection', (ws: WebSocket) => {
    activeTerminals.add(ws);

    ws.on('message', (message: string) => {
      try {
        const payload = JSON.parse(message);
        // Broadcaster for Real-Time UI Mutation Frame Frames
        if (payload.type === "ACCESSIBILITY_MUTATION" || payload.type === "THEME_SYNC") {
          activeTerminals.forEach((client) => {
            if (client !== ws && client.readyState === WebSocket.OPEN) {
              client.send(JSON.stringify({
                type: "GLOBAL_MUTATION_PROPAGATION",
                themeConfig: payload.themeConfig,
                timestamp: Date.now()
              }));
            }
          });
        }
      } catch (err) {
        console.error("Invalid state transaction payload parsed over Socket layer.");
      }
    });

    ws.on('close', () => activeTerminals.delete(ws));
  });
}
