FILL LEVEL SYSTEM
const getFillLevel = (glass) => {
  switch(glass) {
    case "rocks": return "55%";
    case "coupe": return "70%";
    case "highball": return "85%";
    case "snifter": return "60%";
    case "pint": return "90%";
    case "mug": return "95%";
  }
};
