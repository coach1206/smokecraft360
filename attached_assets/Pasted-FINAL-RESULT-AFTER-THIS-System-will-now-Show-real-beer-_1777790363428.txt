🔥 2. AI → IMAGE MATCHING ENGINE
/server/services/imageEngine.ts
import { imageLibrary } from "../data/imageLibrary";

export function getImageForRecommendation(rec: any) {
  if (rec.type === "beer") {
    return random(imageLibrary.beer[rec.subType]);
  }

  if (rec.type === "cigar") {
    return random(imageLibrary.cigar[rec.strength]);
  }

  if (rec.type === "food") {
    return random(imageLibrary.food[rec.category]);
  }

  return "/images/default.jpg";
}

function random(arr: string[]) {
  return arr[Math.floor(Math.random() * arr.length)];
}
🔥 3. CONTEXT ENGINE (TIME + MOOD VISUAL SHIFT)

export function adjustImageByContext(image: string, context: any) {

  if (context.time === "night") {
    return image + "?overlay=dark";
  }

  if (context.mood === "premium") {
    return image + "?filter=gold";
  }

  if (context.weather === "hot") {
    return image + "?brightness=1.2";
  }

  return image;
}

export function syncImagesWithInventory(menu, inventory) {

  return menu.map(item => {

    const available = inventory[item.name]?.available;

    return {
      ...item,
      image: available ? item.image : "/images/sold-out.jpg",
      disabled: !available
    };
  });
}

import { Router } from "express";
import { getImageForRecommendation } from "../services/imageEngine";

const router = Router();

🔥 6. FRONTEND HOOK (REAL-TIME IMAGE)
/client/hooks/useDynamicImage.ts
import { useEffect, useState } from "react";

export function useDynamicImage(recommendation, context) {

  const [image, setImage] = useState("");

  useEffect(() => {

    fetch("/api/images/resolve", {
      method: "POST",
      body: JSON.stringify({ recommendation, context }),
    })
      .then(res => res.json())
      .then(data => setImage(data.image));

  }, [recommendation, context]);

  return image;
}
