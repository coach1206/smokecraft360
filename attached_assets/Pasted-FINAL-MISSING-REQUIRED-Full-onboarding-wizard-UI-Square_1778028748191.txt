import crypto from "crypto";import { logSystemError } from "../errorLogger";const SQUARE_BASE_URL = "https://connect.squareupsandbox.com/v2";export async function createSquareSandboxPayment(order: {  amount: number;  currency?: string;  note?: string;}) {  try {    const accessToken = process.env.SQUARE_ACCESS_TOKEN;    const locationId = process.env.SQUARE_LOCATION_ID;    if (!accessToken || !locationId) {      return {        success: true,        sandboxFallback: true,        transactionId: `square_sandbox_mock_${Date.now()}`,        status: "COMPLETED",        message: "Square sandbox env missing. Mock sandbox transaction created.",      };    }    const body = {      source_id: "cnon:card-nonce-ok",      idempotency_key: crypto.randomUUID(),      amount_money: {        amount: Math.round(order.amount * 100),        currency: order.currency || "USD",      },      location_id: locationId,      note: order.note || "Axiom OS Square sandbox payment",    };    const response = await fetch(`${SQUARE_BASE_URL}/payments`, {      method: "POST",      headers: {        "Square-Version": "2024-12-18",        Authorization: `Bearer ${accessToken}`,        "Content-Type": "application/json",      },      body: JSON.stringify(body),    });    const data = await response.json();    if (!response.ok) {      await logSystemError({        source: "squareSandboxAdapter",        message: "Square sandbox payment failed",        severity: "WARNING",        metadata: data,      });      return {        success: false,        error: data,      };    }    return {      success: true,      transactionId: data.payment?.id,      status: data.payment?.status,      response: data,    };  } catch (error: any) {    await logSystemError({      source: "squareSandboxAdapter",      message: error.message,      stack: error.stack,    });    return {      success: false,      error: error.message,    };  }}
server/services/pos/posRouter.ts
import { db } from "../../db";import { posTransactions } from "../../db/schema/posTransactions";import { createSquareSandboxPayment } from "./squareSandboxAdapter";import { logEvent } from "../../utils/logEvent";export async function processPOSPayment(order: {  venueId: string;  orderId: string;  amount: number;  posMode: string;  sessionId?: string;}) {  if (order.posMode === "off") {    return {      success: false,      error: "POS is disabled",    };  }  let result: any;  if (order.posMode === "square_sandbox") {    result = await createSquareSandboxPayment({      amount: order.amount,      note: `Axiom OS order ${order.orderId}`,    });  } else if (order.posMode === "stripe_fallback") {    result = {      success: true,      transactionId: `stripe_demo_${Date.now()}`,      status: "COMPLETED",    };  } else {    result = {      success: true,      transactionId: `${order.posMode}_${Date.now()}`,      status: "COMPLETED",    };  }  await db.insert(posTransactions).values({    venueId: order.venueId,    orderId: order.orderId,    posProvider: order.posMode.includes("square") ? "SQUARE" : "DEMO",    posMode: order.posMode,    externalTransactionId: result.transactionId || null,    amount: String(order.amount),    status: result.success ? "COMPLETED" : "FAILED",    requestPayload: order,    responsePayload: result,    isDemo: true,  });  if (order.sessionId) {    await logEvent({      venueId: order.venueId,      sessionId: order.sessionId,      eventType: "POS_SANDBOX_PAYMENT_CREATED",      price: order.amount,      metadata: {        posMode: order.posMode,        transactionId: result.transactionId,      },    });  }  return result;}
