BackButton (GLOBAL)
import { useNavigate } from "react-router-dom";

export default function BackButton() {
  const navigate = useNavigate();

  return (
    <button onClick={() => navigate(-1)} style={{
      position: "absolute",
      top: "20px",
      left: "20px",
      background: "rgba(0,0,0,0.7)",
      color: "#fff",
      padding: "12px"
    }}>
      Back
    </button>
  );
}
