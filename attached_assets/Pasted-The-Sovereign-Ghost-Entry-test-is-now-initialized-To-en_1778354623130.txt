import React from 'react';
import { useUser } from '../context/UserContext';
import { AccessLevel, checkAccess } from '../../services/AuthorityEngine';

export const PermissionGate = ({ children, requiredLevel, mask = true }) => {
  const { currentUser } = useUser();
  const hasAccess = checkAccess(currentUser.tier, requiredLevel);

  if (!hasAccess) {
    // Hidden features disappear entirely if 'mask' is true
    return mask ? null : <div className="grayscale opacity-10 pointer-events-none">{children}</div>;
  }

  return <>{children}</>;
};
