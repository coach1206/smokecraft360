import {
  pgTable,
  uuid,
  varchar,
  boolean,
  timestamp,
  text,
  integer,
  jsonb,
} from "drizzle-orm/pg-core";

export const venueAiProviders = pgTable("venue_ai_providers", {
  id: uuid("id").defaultRandom().primaryKey(),

  venueId: uuid("venue_id").notNull(),

  provider: varchar("provider", {
    length: 100,
  }).notNull(), // openai | anthropic | gemini

  mode: varchar("mode", {
    length: 50,
  }).notNull(), // managed | byok

  isActive: boolean("is_active").default(true),

  createdAt: timestamp("created_at").defaultNow(),
});

export const venueApiKeys = pgTable("venue_api_keys", {
  id: uuid("id").defaultRandom().primaryKey(),

  venueId: uuid("venue_id").notNull(),

  provider: varchar("provider", {
    length: 100,
  }).notNull(),

  encryptedKey: text("encrypted_key").notNull(),

  createdAt: timestamp("created_at").defaultNow(),
});

export const venueAiUsage = pgTable("venue_ai_usage", {
  id: uuid("id").defaultRandom().primaryKey(),

  venueId: uuid("venue_id").notNull(),

  provider: varchar("provider", {
    length: 100,
  }).notNull(),

  tokensUsed: integer("tokens_used").default(0),

  requests: integer("requests").default(0),

  estimatedCost: integer("estimated_cost").default(0),

  createdAt: timestamp("created_at").defaultNow(),
});

export const venueAiBillingModes = pgTable(
  "venue_ai_billing_modes",
  {
    id: uuid("id").defaultRandom().primaryKey(),

    venueId: uuid("venue_id").notNull(),

    billingMode: varchar("billing_mode", {
      length: 50,
    }).notNull(), // included | byok

    packageTier: varchar("package_tier", {
      length: 50,
    }).notNull(),

    createdAt: timestamp("created_at").defaultNow(),
  }
);

export const venueAiLimits = pgTable("venue_ai_limits", {
  id: uuid("id").defaultRandom().primaryKey(),

  venueId: uuid("venue_id").notNull(),

  monthlyTokenLimit: integer(
    "monthly_token_limit"
  ).default(100000),

  currentUsage: integer("current_usage").default(0),

  hardStop: boolean("hard_stop").default(false),

  createdAt: timestamp("created_at").defaultNow(),
});

type AIProvider =
  | "openai"
  | "anthropic"
  | "gemini";

interface RouteRequest {
  provider: AIProvider;
  prompt: string;
  apiKey?: string;
}

export class AIRouter {
  async route({
    provider,
    prompt,
    apiKey,
  }: RouteRequest) {
    switch (provider) {
      case "openai":
        return this.handleOpenAI(prompt, apiKey);

      case "anthropic":
        return this.handleAnthropic(prompt, apiKey);

      case "gemini":
        return this.handleGemini(prompt, apiKey);

      default:
        throw new Error(
          "Unsupported AI provider"
        );
    }
  }

  private async handleOpenAI(
    prompt: string,
    apiKey?: string
  ) {
    return {
      provider: "openai",
      success: true,
      response:
        "OpenAI response placeholder",
    };
  }

  private async handleAnthropic(
    prompt: string,
    apiKey?: string
  ) {
    return {
      provider: "anthropic",
      success: true,
      response:
        "Anthropic response placeholder",
    };
  }

  private async handleGemini(
    prompt: string,
    apiKey?: string
  ) {
    return {
      provider: "gemini",
      success: true,
      response:
        "Gemini response placeholder",
    };
  }
}

export const aiRouter = new AIRouter();
