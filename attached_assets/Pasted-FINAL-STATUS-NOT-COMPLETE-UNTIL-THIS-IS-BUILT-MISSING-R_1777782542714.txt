function calculateScore(user, cigar) {
  let score = 50;

  user.flavors.forEach(f => {
    if (cigar.flavors.includes(f)) score += 8;
    else score -= 4;
  });

  if (user.strength === cigar.strength) score += 10;
  else score -= 6;

  if (user.mood === cigar.mood) score += 6;
  else score -= 3;

  if (user.flavors.includes("floral") && user.flavors.includes("smoky")) {
    score += 10;
  }

  return Math.max(40, Math.min(score, 98));
}

function calculateImprovement(current, history) {
  const last = history.slice(-5);
  const avg = last.reduce((a, b) => a + b, 0) / last.length || 50;
  return ((current - avg) / avg) * 100;
}

function getArchetype(user) {
  if (user.flavors.includes("smoky") && user.flavors.includes("leather"))
    return "Dominant Palate";
  if (user.flavors.includes("creamy"))
    return "Smooth Operator";
  if (user.flavors.length > 3)
    return "Risk Taker";
  return "Craftsman";
}
