// 2. THE DISTRIBUTION BUNDLER (Logic for Zip/Email Export)
export const prepareManufacturerBundle = (orderData: { manufacturer: string, qty: number, type: string }) => {
  const batchId = `TITAN-V-${Math.random().toString(36).substring(2, 7).toUpperCase()}`;
  
  // This JSON file gets baked into the Zip for the manufacturer
  const deploymentManifest = {
    ...orderData,
    batchId,
    owner: SOVEREIGN_ROOT.entity,
    isActivated: false, // LOCKS THE HARDWARE BY DEFAULT
    timestamp: new Date().toISOString()
  };

  return {
    filename: `TITAN_V_DISTRO_${batchId}.zip`,
    manifest: deploymentManifest
  };
};
