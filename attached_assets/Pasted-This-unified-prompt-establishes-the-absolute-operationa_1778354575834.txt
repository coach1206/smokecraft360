1. The Authority Engine (server/services/AuthorityEngine.ts)
This handles the "who sees what" logic.

TypeScript
// Sovereign Access Levels
export enum AccessLevel {
  SOVEREIGN = 0, // God Mode
  SHIFT_LEAD = 1, // Operational Admin
  MENTOR = 2      // Guest Experience Only
}

export interface UserProfile {
  id: string;
  name: string;
  tier: AccessLevel;
  permissions: string[]; // e.g., ['inventory_edit', 'kill_switch']
}

export const checkAccess = (user: UserProfile, requiredLevel: AccessLevel): boolean => {
  return user.tier <= requiredLevel;
};

3. The Permissions Gate (client/src/components/admin/PermissionGate.tsx)
This "Masks" features based on the user's level.

TypeScript
import React from 'react';
import { useUser } from '../context/UserContext';
import { AccessLevel, checkAccess } from '../../services/AuthorityEngine';

export const PermissionGate = ({ children, requiredLevel, mask = true }) => {
  const { currentUser } = useUser();
  const hasAccess = checkAccess(currentUser, requiredLevel);

  if (!hasAccess) {
    // If 'mask' is true, the component is invisible. If false, it could show a locked state.
    return mask ? null : <div className="grayscale opacity-20 pointer-events-none">{children}</div>;
  }

  return <>{children}</>;
};
