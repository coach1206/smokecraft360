import React, { useState, useEffect } from 'react';
import { StyleSheet, View, Text, TouchableOpacity, ScrollView, Dimensions, Animated, PanResponder } from 'react-native';

const { width, height } = Dimensions.get('window');

// Strongly Typed Payload Frameworks
interface POSItem {
  id: string;
  name: string;
  price: number;
  category: 'bar' | 'humidor' | 'kitchen';
  isLocked?: boolean;
}

interface MemberLockerProfile {
  lockerId: string;
  name: string;
  preferredCut: string;
  preferredLight: string;
  stickInventory: string[];
}

// Sample Static Data Mapping
const HIGH_VELOCITY_WELLS: POSItem[] = [
  { id: 'inv_4', name: 'Macallan 25 Year Pour', price: 250.00, category: 'bar' },
  { id: 'well_2', name: 'House Old Fashioned', price: 18.00, category: 'bar' },
  { id: 'well_3', name: 'Reposado Tequila Neat', price: 16.00, category: 'bar' },
  { id: 'well_4', name: 'Premium Craft Draught', price: 9.00, category: 'bar' },
];

const EXTENDED_ASSETS: POSItem[] = [
  { id: 'inv_1', name: 'Arturo Fuente Opus X', price: 45.00, category: 'humidor' },
  { id: 'inv_2', name: 'Davidoff Late Hour Churchill', price: 38.00, category: 'humidor' },
  { id: 'inv_3', name: 'Wagyu Strip Steak A5', price: 120.00, category: 'kitchen', isLocked: true },
];

const ACTIVE_MEMBER: MemberLockerProfile = {
  lockerId: 'L-42',
  name: 'John Manuel',
  preferredCut: 'V-Cut',
  preferredLight: 'Cedar Spill',
  stickInventory: ['Opus X x 3', 'Padron 1926 x 5']
};

export default function TouchPOS3App() {
  // System State Observables
  const [activeTicket, setActiveTicket] = useState<POSItem[]>([]);
  const [isLockerDrawerOpen, setIsLockerDrawerOpen] = useState(false);
  const [optimisticLockedId, setOptimisticLockedId] = useState<string | null>(null);
  const [networkMode, setNetworkMode] = useState<'CLOUD' | 'SHADOW'>('CLOUD');
  
  // Slide Drawer Animation Node
  const drawerAnimationValue = useState(new Animated.Value(width))[0];

  // Toggle Humidor/Member Glass Drawer with Weighted Damping Kinematics
  const toggleMemberDrawer = (open: boolean) => {
    setIsLockerDrawerOpen(open);
    Animated.spring(drawerAnimationValue, {
      toValue: open ? width * 0.35 : width,
      useNativeDriver: true,
      bounciness: 4,
      speed: 12,
    }).start();
  };

  // Simulate Production API Socket.IO TOUCH_LOCK TTL Hook
  const handleHighVelocityTap = (item: POSItem) => {
    if (item.isLocked || optimisticLockedId) return;

    // Trigger Optimistic Frontend Lock State instantly upon touch contact
    setOptimisticLockedId(item.id);

    // Simulate standard production Socket.IO TTL timeout release loop
    setTimeout(() => {
      setActiveTicket(prev => [...prev, item]);
      setOptimisticLockedId(null); // Auto-release lock (TOUCH_UNLOCK frame)
    }, 250); // Fast 250ms infrastructure processing turn-around
  };

  const calculateTotal = () => activeTicket.reduce((acc, item) => acc + item.price, 0);

  return (
    <View style={styles.canvasContainer}>
      
      {/* HEADER NAVIGATION BAR LAYOUT */}
      <View style={styles.headerStatusBar}>
        <View>
          <Text style={styles.platformBrandMark}>NOVEE OS</Text>
          <Text style={styles.operatorSessionLabel}>FOH CONSOLE // STAFF_JC_01</Text>
        </View>
        
        <TouchableOpacity 
          style={[styles.networkBadgeFrame, { borderColor: networkMode === 'CLOUD' ? '#D4AF37' : '#FF3B30' }]}
          onPress={() => setNetworkMode(prev => prev === 'CLOUD' ? 'SHADOW' : 'CLOUD')}
        >
          <Text style={[styles.networkBadgeText, { color: networkMode === 'CLOUD' ? '#D4AF37' : '#FF3B30' }]}>
            {networkMode === 'CLOUD' ? 'E.A.T. SYNCHRONIZED' : 'SHADOW MODE ACTIVE'}
          </Text>
        </TouchableOpacity>
      </View>

      {/* CORE WORKSPACE PANELS */}
      <View style={styles.mainWorkspaceLayout}>
        
        {/* LEFT COLUMN: HIGH-VELOCITY FAST BAR WELL (40% VIEWPORT WIDTH) */}
        <View style={styles.fastBarWellColumn}>
          <Text style={styles.sectionClusterHeading}>High-Velocity Bar Well</Text>
          <View style={styles.fastBarGrid}>
            {HIGH_VELOCITY_WELLS.map((wellItem) => {
              const isLockedByTTL = optimisticLockedId === wellItem.id;
              return (
                <TouchableOpacity
                  key={wellItem.id}
                  style={[styles.touchTargetItemButton, isLockedByTTL && styles.itemButtonOptimisticLocked]}
                  onPress={() => handleHighVelocityTap(wellItem)}
                  activeOpacity={0.6}
                >
                  <Text style={styles.wellItemTextName}>{wellItem.name}</Text>
                  <Text style={styles.wellItemTextPrice}>${wellItem.price.toFixed(2)}</Text>
                  {isLockedByTTL && <View style={styles.ttlProcessingOverlay} />}
                </TouchableOpacity>
              );
            })}
          </View>

          <Text style={styles.sectionClusterHeading}>Extended Venue Assets</Text>
          <ScrollView contentContainerStyle={styles.extendedAssetStack}>
            {EXTENDED_ASSETS.map((assetItem) => {
              const is86ed = assetItem.isLocked;
              return (
                <TouchableOpacity
                  key={assetItem.id}
                  style={[styles.extendedRowButton, is86ed && styles.extendedRowButton86ed]}
                  onPress={() => handleHighVelocityTap(assetItem)}
                  disabled={is86ed}
                >
                  <View>
                    <Text style={[styles.extendedRowName, is86ed && styles.textMuted86]}>{assetItem.name}</Text>
                    <Text style={styles.extendedRowMeta}>{assetItem.category.toUpperCase()} STATE</Text>
                  </View>
                  {is86ed ? (
                    <Text style={styles.badgeLockedText86}>86'ED</Text>
                  ) : (
                    <Text style={styles.extendedRowPrice}>${assetItem.price.toFixed(2)}</Text>
                  )}
                </TouchableOpacity>
              );
            })}
          </ScrollView>
        </View>

        {/* RIGHT COLUMN: CURRENT ACTIVE TICKET RUNNER & CHECK OVERVIEW (60% VIEWPORT WIDTH) */}
        <View style={styles.activeCheckColumn}>
          <View style={styles.checkHeaderRow}>
            <Text style={styles.sectionClusterHeading}>Active Check — Table 03</Text>
            <TouchableOpacity style={styles.memberProfileTriggerButton} onPress={() => toggleMemberDrawer(true)}>
              <Text style={styles.memberTriggerText}>VIEW MEMBER LOCKER</Text>
            </TouchableOpacity>
          </View>

          <ScrollView style={styles.activeItemsScrollArea}>
            {activeTicket.length === 0 ? (
              <View style={styles.emptyCheckStateFrame}>
                <Text style={styles.emptyCheckStateText}>Terminal context clear. Tap assets to fire order.</Text>
              </View>
            ) : (
              activeTicket.map((ticketItem, index) => (
                <View key={index} style={styles.ticketItemRowLine}>
                  <Text style={styles.ticketLineItemName}>{ticketItem.name}</Text>
                  <Text style={styles.ticketLineItemPrice}>${ticketItem.price.toFixed(2)}</Text>
                </View>
              ))
            )}
          </ScrollView>

          {/* LOWER TRANSACTION REVENUE CONTROL BLOCK */}
          <View style={styles.transactionControlFooter}>
            <View style={styles.revenueSummaryRow}>
              <Text style={styles.revenueTotalLabel}>Ledger Total Balance</Text>
              <Text style={styles.revenueTotalValue}>${calculateTotal().toFixed(2)}</Text>
            </View>
            <View style={styles.actionButtonGroupRow}>
              <TouchableOpacity style={styles.splitBillButtonTouch} onPress={() => alert('3-Tap Split Engine Engaged')}>
                <Text style={styles.splitButtonLabel}>SPLIT CHECK</Text>
              </TouchableOpacity>
              <TouchableOpacity 
                style={styles.settleBillButtonTouch} 
                onPress={() => {
                  alert('Ledger Settled Globally');
                  setActiveTicket([]);
                }}
              >
                <Text style={styles.settleButtonLabel}>SETTLE TRANSACTION</Text>
              </TouchableOpacity>
            </View>
          </View>
        </View>

      </View>

      {/* TRANSFLUENT MEMBER PROFILE & HUMIDOR STATUS DRAWER FRAME */}
      <Animated.View style={[styles.obsidianMemberDrawerPanel, { transform: [{ translateX: drawerAnimationValue }] }]}>
        <View style={styles.drawerHeaderFrame}>
          <Text style={styles.drawerHeaderTitle}>HUMIDOR MEMBER LOG</Text>
          <TouchableOpacity style={styles.drawerCloseButtonTouch} onPress={() => toggleMemberDrawer(false)}>
            <Text style={styles.drawerCloseButtonText}>✕ CLOSE</Text>
          </TouchableOpacity>
        </View>

        <ScrollView style={styles.drawerContentScrollBody}>
          <Text style={styles.drawerProfileNameText}>{ACTIVE_MEMBER.name}</Text>
          <Text style={styles.drawerProfileLockerIdLabel}>Humidor Asset Container: {ACTIVE_MEMBER.lockerId}</Text>
          
          <View style={styles.drawerMetadataDividerLine} />
          
          <Text style={styles.drawerSubSectionHeading}>PREFERENCE MATRIX</Text>
          <Text style={styles.drawerMetadataBodyText}>Ritual Cut Specification: {ACTIVE_MEMBER.preferredCut}</Text>
          <Text style={styles.drawerMetadataBodyText}>Sensory Ignition Routine: {ACTIVE_MEMBER.preferredLight}</Text>

          <View style={styles.drawerMetadataDividerLine} />

          <Text style={styles.drawerSubSectionHeading}>CURRENT LOCKER TRACKING BALANCE</Text>
          {ACTIVE_MEMBER.stickInventory.map((stick, i) => (
            <View key={i} style={styles.drawerInventoryRowItem}>
              <Text style={styles.drawerInventoryItemText}>{stick}</Text>
            </View>
          ))}
        </ScrollView>
      </Animated.View>

    </View>
  );
}
