PART 2: IMAGE FETCH FUNCTION

async function fetchImage(query) {
const res = await fetch(https://api.unsplash.com/search/photos?query=${query}&client_id=YOUR_KEY);
const data = await res.json();
return data.results[0]?.urls?.regular;
}

async function getFlavorImage(flavor) {
const queries = FLAVOR_IMAGE_MAP[flavor];
const randomQuery = queries[Math.floor(Math.random() * queries.length)];
return await fetchImage(randomQuery);
}

function matchCigar(userSelections) {
return CIGAR_DATABASE.map(cigar => {
let score = 0;

userSelections.forEach(sel => {
  if (cigar.flavors.includes(sel)) score += 2;
  if (cigar.strength === sel) score += 1;
});

return { cigar, score };

}).sort((a, b) => b.score - a.score)[0].cigar;
}
