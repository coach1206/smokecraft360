// server/db/schema/dataEngine.ts
import {
  pgTable,
  text,
  timestamp,
  jsonb,
  uuid,
  numeric,
} from "drizzle-orm/pg-core";

export const userSessions = pgTable("user_sessions", {
  sessionId: text("session_id").primaryKey(),
  venueId: text("venue_id").notNull(),
  deviceId: text("device_id"),
  startedAt: timestamp("started_at").defaultNow(),
  endedAt: timestamp("ended_at"),
});

export const eventLogs = pgTable("event_logs", {
  id: uuid("id").primaryKey().defaultRandom(),
  venueId: text("venue_id").notNull(),
  sessionId: text("session_id").notNull(),
  eventType: text("event_type").notNull(),
  productId: text("product_id"),
  category: text("category"),
  price: numeric("price"),
  metadata: jsonb("metadata"),
  timestamp: timestamp("timestamp").defaultNow(),
});
2. EVENT LOGGER
// server/utils/logEvent.ts
import { db } from "../db";
import { eventLogs } from "../db/schema/dataEngine";

export async function logEvent(event: any) {
  try {
    await db.insert(eventLogs).values({
      venueId: event.venueId,
      sessionId: event.sessionId,
      eventType: event.eventType,
      productId: event.productId || null,
      category: event.category || null,
      price: event.price || null,
      metadata: event.metadata || {},
      timestamp: new Date(),
    });
  } catch (err) {
    console.error("Event logging failed:", err);
  }
}

// server/routes/sessionRoutes.ts
import express from "express";
import { db } from "../db";
import { userSessions } from "../db/schema/dataEngine";
import { nanoid } from "nanoid";

const router = express.Router();

router.post("/start", async (req, res) => {
  const sessionId = nanoid();

  await db.insert(userSessions).values({
    sessionId,
    venueId: req.body.venueId,
    deviceId: req.body.deviceId,
  });

  res.json({ sessionId });
});

export default router;

5. AGGREGATION SERVICE
// server/services/aggregationService.ts
import { db } from "../db";
import { eventLogs } from "../db/schema/dataEngine";
import { sql } from "drizzle-orm";

export async function getTopProducts(venueId: string) {
  return db.execute(sql`
    SELECT product_id, COUNT(*) as total
    FROM event_logs
    WHERE venue_id = ${venueId}
      AND event_type = 'PRODUCT_PURCHASED'
    GROUP BY product_id
    ORDER BY total DESC
    LIMIT 10
  `);
}
6. ANALYTICS ROUTES
// server/routes/analyticsRoutes.ts
import express from "express";
import { requireFeature } from "../middleware/requireFeature";
import { getTopProducts } from "../services/aggregationService";

const router = express.Router();

router.get(
  "/top-products",
  requireFeature("ADVANCED_ANALYTICS"),
  async (req, res) => {
    const data = await getTopProducts(req.query.venueId as string);
    res.json(data);
  }
);

export default router;

    </div>
  );
}
