# 6. SESSION SNAPSHOT BACKEND
### `server/services/SessionSnapshotService.ts`

```ts
interface Snapshot {
  sessionId: string;
  route: string;
  xp: number;
  mentor: string;
  challengeStep: number;
}

const sessionSnapshots = new Map<string, Snapshot>();

export class SessionSnapshotService {

  static save(snapshot: Snapshot) {
    sessionSnapshots.set(snapshot.sessionId, snapshot);
  }

  static restore(sessionId: string) {
    return sessionSnapshots.get(sessionId);
  }

}
```

# 7. HANDOFF ROUTES
### `server/routes/handoff.ts`

```ts
import express from "express";
import { SessionSnapshotService } from "../services/SessionSnapshotService";

const router = express.Router();

router.post("/handoff/start", async (req, res) => {

  const {
    sessionId,
    snapshot
  } = req.body;

  SessionSnapshotService.save({
    sessionId,
    ...snapshot
  });

  return res.json({
    success: true,
    mode: "operational"
  });
});

router.post("/handoff/return", async (req, res) => {

  const { sessionId } = req.body;

  const snapshot =
    SessionSnapshotService.restore(sessionId);

  return res.json({
    success: true,
    snapshot
  });
});

export default router;
```

# 8. RESUME EXPERIENCE BUTTON LOGIC
### `client/src/hooks/useResumeExperience.ts`

```tsx
import axios from "axios";

export async function resumeExperience(sessionId: string) {

  const res = await axios.post(
    "/api/handoff/return",
    { sessionId }
  );

  return res.data.snapshot;
}
```
