import React, { useEffect, useState } from 'react';

interface HeroProps {
    activeCigarSku: string;
    activeLiquorSku: string;
    wsNetworkStreamUrl: string;
}

export const HeroPairingRenderer: React.FC<HeroProps> = ({ activeCigarSku, activeLiquorSku, wsNetworkStreamUrl }) => {
    const [canvasUrl, setCanvasUrl] = useState<string>("/assets/curated_cigar_bourbon_portrait.jpg");
    const [isRegenerating, setIsRegenerating] = useState<boolean>(false);

    useEffect(() => {
        const wsConnection = new WebSocket(wsNetworkStreamUrl);
        wsConnection.onmessage = (event) => {
            const incomingFrame = JSON.parse(event.data);
            if (incomingFrame.channel === "PAIRING_ALTERATION" && incomingFrame.data.composition.cinematic_background_canvas) {
                setIsRegenerating(true);
                setCanvasUrl(incomingFrame.data.composition.cinematic_background_canvas);
                setIsRegenerating(false);
            }
        };
        return () => wsConnection.close();
    }, [wsNetworkStreamUrl]);

    return (
        <div className="w-full bg-[#F9F8F6] rounded-xs p-6 flex flex-col gap-6 border border-[#2C2C30]/10 shadow-sm">
            {/* Core Vertical 3-Column Split Display Screen */}
            <div className="grid grid-cols-12 gap-6 items-stretch">
                
                {/* Column Node A: Tall Portrait Primary Image Canvas */}
                <div className="col-span-6 relative h-[480px] rounded-xs overflow-hidden shadow-md bg-[#010101]">
                    <div 
                        className={`absolute inset-0 bg-cover bg-center transition-all duration-[2000ms] ease-in-out ${isRegenerating ? 'opacity-30 filter blur-sm scale-105' : 'opacity-100 scale-100'}`}
                        style={{ backgroundImage: `url('${canvasUrl}')` }}
                    />
                    <div className="absolute inset-0 bg-gradient-to-t from-[#010101]/80 via-transparent to-transparent" />
                    
                    {/* Live Volumetric Smoke Overlay Simulation Loop */}
                    <div 
                        className="absolute inset-0 mix-blend-screen pointer-events-none opacity-20 bg-cover bg-center"
                        style={{ backgroundImage: `url('/assets/fx/ambient_volumetric_smoke_loop.gif')` }}
                    />
                </div>

                {/* Column Node B: Curation Product Details & Numeric Ratings */}
                <div className="col-span-6 flex flex-col justify-between py-2">
                    <div className="flex flex-col gap-4">
                        <div className="flex flex-col gap-1">
                            <span className="text-[11px] font-mono text-[#D4AF37] tracking-[2px] uppercase">Active Connoisseur Selection</span>
                            <h2 className="text-3xl font-semibold text-[#010101] tracking-tight">Rocky Patel Vintage 1992</h2>
                            <p className="text-sm font-mono text-[#8E8E93]">Maduro · Honduras / Nicaragua Wrapper</p>
                        </div>
                        <div className="h-[1px] w-full bg-[#2C2C30]/10" />
                        <p className="text-sm text-[#2C2C30] leading-relaxed">
                            A rich, full-bodied masterwork aged for 10 structural years. Delivers intense layers of dark cocoa, charred cedar wood, peat loam, and robust espresso bean undertones.
                        </p>
                    </div>

                    {/* Quality Ratings Section (No Font Character Star Glyphs Allowed) */}
                    <div className="flex items-center gap-4 bg-[#F4F3EF] p-4 rounded-xs border border-[#2C2C30]/10">
                        <div className="flex flex-col">
                            <span className="text-[10px] font-mono text-[#8E8E93] uppercase">Sovereign Index Profile</span>
                            <div className="text-xl font-mono font-bold text-[#010101]">94 / 100 POINT EVALUATION</div>
                        </div>
                    </div>
                </div>
            </div>

            {/* Column Node C: Bottom Thumbnail Preview Strip Matrix */}
            <div className="flex flex-col gap-3 border-t border-[#2C2C30]/10 pt-4">
                <span className="text-[10px] font-mono text-[#8E8E93] tracking-widest uppercase">Alternative Sensory Profiles</span>
                <div className="grid grid-cols-3 gap-4 overflow-x-hidden">
                    <div className="h-24 rounded-xs overflow-hidden bg-cover bg-center border border-[#2C2C30]/10" style={{ backgroundImage: `url('/assets/preview_lounge_interior.jpg')` }} />
                    <div className="h-24 rounded-xs overflow-hidden bg-cover bg-center border border-[#2C2C30]/10" style={{ backgroundImage: `url('/assets/preview_spirit_pour.jpg')` }} />
                    <div className="h-24 rounded-xs overflow-hidden bg-cover bg-center border border-[#2C2C30]/10" style={{ backgroundImage: `url('/assets/preview_cigar_detail.jpg')` }} />
                </div>
            </div>
        </div>
    );
};
