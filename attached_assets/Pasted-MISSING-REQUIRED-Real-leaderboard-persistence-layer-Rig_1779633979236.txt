  },

  {
    name: "AFICIONADO",
    minXp: 25000,
    maxXp: 49999
  },

  {
    name: "MASTER OF THE LEAF",
    minXp: 50000,
    maxXp: 999999
  }
];

export function determineRank(xp: number) {
  return RANKS.find(
    rank =>
      xp >= rank.minXp &&
      xp <= rank.maxXp
  );
}
XP ENGINE
server/services/xp-engine.ts
import { db } from "../db";
import {
  users,
  xpTransactions
} from "../db/schema";

import { eq } from "drizzle-orm";

import { determineRank } from "./rank-engine";

export async function awardXp(
  userId: string,
  amount: number,
  action: string
) {

  const user = await db.query.users.findFirst({
    where: eq(users.id, userId)
  });

  if (!user) {
    throw new Error("User not found");
  }

  const newXp = user.totalXp + amount;

  const rank = determineRank(newXp);

  await db.insert(xpTransactions).values({
    userId,
    xpEarned: amount,
    action
  });

  await db
    .update(users)
    .set({
      totalXp: newXp,
      currentRank: rank?.name
    })
    .where(eq(users.id, userId));

  return {
    success: true,
    newXp,
    rank: rank?.name
  };
}
LEADERBOARD ENGINE
server/services/leaderboard-engine.ts
import { db } from "../db";

export async function getLeaderboard() {

  const leaderboard = await db.query.users.findMany({
    orderBy: (users, { desc }) => [
      desc(users.totalXp)
    ],
    limit: 50
  });

  return leaderboard.map((user, index) => ({
    rankPosition: index + 1,

    username: user.username,

    totalXp: user.totalXp,

    currentRank: user.currentRank,

    avatar: user.avatar
  }));
}
REALTIME SOCKET SYSTEM
server/sockets/leaderboard-socket.ts
import { Server } from "socket.io";

import { getLeaderboard }
from "../services/leaderboard-engine";

export function initializeLeaderboardSocket(
  io: Server
) {

  io.on("connection", socket => {

    socket.on(
      "requestLeaderboard",
      async () => {

        const board =
          await getLeaderboard();

        socket.emit(
          "leaderboardUpdate",
          board
        );
      }
    );
  });
}
LEADERBOARD ROUTES
server/routes/leaderboard-routes.ts
import express from "express";

import {
  getLeaderboard
} from "../services/leaderboard-engine";

const router = express.Router();

router.get("/", async (_, res) => {

  try {

    const board =
      await getLeaderboard();

    return res.json({
      success: true,
      leaderboard: board
    });

  } catch (err) {

    return res.status(500).json({
      success: false
    });
  }
});

export default router;
CHALLENGE COMPLETION ROUTE
server/routes/challenge-routes.ts
import express from "express";

import { awardXp }
from "../services/xp-engine";

const router = express.Router();

router.post(
  "/complete",
  async (req, res) => {

    try {

      const {
        userId,
        xp,
        challengeName
      } = req.body;

      const result =
        await awardXp(
          userId,
          xp,
          challengeName
        );

      return res.json({
        success: true,
        result
      });

    } catch (err) {

      return res.status(500).json({
        success: false
      });
    }
  }
);

export default router;
MAIN SERVER
server/server.ts
import express from "express";

import http from "http";

import { Server } from "socket.io";

import leaderboardRoutes
from "./routes/leaderboard-routes";

import challengeRoutes
from "./routes/challenge-routes";

import {
  initializeLeaderboardSocket
} from "./sockets/leaderboard-socket";

const app = express();

app.use(express.json());

app.use(
  "/api/leaderboard",
  leaderboardRoutes
);

app.use(
  "/api/challenges",
  challengeRoutes
);

const server = http.createServer(app);

const io = new Server(server, {
  cors: {
    origin: "*"
  }
});

initializeLeaderboardSocket(io);

server.listen(5000, () => {

  console.log(
    "SmokeCraft backend running"
  );
});
