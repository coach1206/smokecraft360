import { useEffect, useRef, useState } from "react";

export default function CraftEngine({ type, config }) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [showReal, setShowReal] = useState(false);

  useEffect(() => {
    const canvas = canvasRef.current!;
    const ctx = canvas.getContext("2d")!;
    let progress = 0;
    let lastFrame = 0;

    const settings = getPhysics(type);

    playSound(type);

    const draw = (timestamp: number) => {
      if (timestamp - lastFrame < 16) {
        requestAnimationFrame(draw);
        return;
      }
      lastFrame = timestamp;

      ctx.clearRect(0, 0, canvas.width, canvas.height);

      if (type === "vape") {
        drawVape(ctx, progress, settings);
      } else {
        drawLiquid(ctx, progress, settings, config);
      }

      progress += settings.speed;

      if (progress < 1) {
        requestAnimationFrame(draw);
      }
    };

    requestAnimationFrame(draw);

    setTimeout(() => setShowReal(true), 2000);

  }, [type, config]);

  return (
    <div className="relative">
      <canvas ref={canvasRef} width={400} height={400} />
      <img src={`/overlays/${type}.png`} className="absolute top-0 w-full pointer-events-none" />
      <img src={config.generatedImage} className={`absolute top-0 w-full transition-opacity duration-700 ${showReal ? "opacity-100" : "opacity-0"}`} />
    </div>
  );
}
💧 LIQUID ENGINE (WITH SWIRL + ICE + FOAM)
function drawLiquid(ctx, progress, settings, config) {
  const { color, fillLevel } = config;

  const liquidHeight = progress * 300 * fillLevel;

  ctx.beginPath();

  for (let x = 100; x <= 300; x++) {
    const swirlOffset = config.swirl ? Math.sin(Date.now() * 0.002 + x * 0.05) * 5 : 0;

    const y =
      350 - liquidHeight +
      Math.sin((x + progress * settings.waveSpeed) * 0.05) *
        settings.waveStrength +
      swirlOffset;

    ctx.lineTo(x, y);
  }

  ctx.lineTo(300, 350);
  ctx.lineTo(100, 350);
  ctx.closePath();

  const gradient = ctx.createLinearGradient(0, 0, 0, 400);
  gradient.addColorStop(0, color);
  gradient.addColorStop(1, "#000");

  ctx.fillStyle = gradient;
  ctx.fill();

  if (config.ice) drawIce(ctx, liquidHeight);

  if (settings.foam) drawFoam(ctx, liquidHeight, progress);
}
🫧 FOAM SYSTEM (RISING + SETTLING)
function drawFoam(ctx, liquidHeight, progress) {
  const foamHeight = Math.min(20 + progress * 30, 40);

  for (let i = 0; i < 80; i++) {
    ctx.beginPath();
    ctx.arc(
      120 + Math.random() * 160,
      350 - liquidHeight - Math.random() * foamHeight,
      Math.random() * 5,
      0,
      Math.PI * 2
    );
    ctx.fillStyle = "rgba(255,255,255,0.7)";
    ctx.fill();
  }
}

function drawVape(ctx, progress, settings) {
  for (let i = 0; i < 40; i++) {
    const x = touchX + Math.sin(i + progress * 6) * 50;
    const y = 300 - progress * 200 + i * 2;

    ctx.beginPath();
    ctx.arc(x, y, settings.density * Math.random() * 10, 0, Math.PI * 2);
    ctx.fillStyle = "rgba(255,255,255,0.06)";
    ctx.fill();
  }
}
⚙️ PHYSICS ENGINE
function getPhysics(type) {
  switch (type) {
    case "beer":
      return { speed: 0.015, waveStrength: 6, waveSpeed: 120, foam: true };
    case "whiskey":
      return { speed: 0.01, waveStrength: 2, waveSpeed: 80, foam: false };
    case "wine":
      return { speed: 0.012, waveStrength: 3, waveSpeed: 90, foam: false };
    case "vape":
      return { speed: 0.02, density: 1.3 };
    default:
      return { speed: 0.01, waveStrength: 3, waveSpeed: 100 };
  }
}
