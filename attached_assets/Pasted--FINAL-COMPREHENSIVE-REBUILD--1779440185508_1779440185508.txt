## SECTION 2: ARCHITECTURAL DATA CORE SCHEMAS (POSTGRESQL)

To handle real-time file uploads, metadata asset tracking, and AI image pipeline references, deploy these relational database structures.

```sql
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 2.1 Media Assets Core System Table
CREATE TABLE media_assets (
    asset_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    file_name VARCHAR(255) NOT NULL,
    storage_url TEXT NOT NULL,
    thumbnail_url TEXT NOT NULL,
    mime_type VARCHAR(64) NOT NULL,
    file_size_bytes INTEGER NOT NULL,
    category_tag VARCHAR(32) CHECK (category_tag IN ('CIGAR', 'BOURBON', 'WHISKEY', 'COCKTAIL', 'FOOD', 'PAIRING_SCENE', 'LOUNGE_ATMOSPHERE', 'STAFF_IMAGERY', 'EVENT_PROMOTION', 'VIP_ROOM', 'LIGHTING_SCENE', 'MUSIC_THEME', 'FLAVOR_VISUAL')),
    flavor_tags VARCHAR(64)[] DEFAULT '{}',
    atmosphere_tags VARCHAR(64)[] DEFAULT '{}',
    metadata_json JSONB DEFAULT '{}'::jsonb,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2.2 Pairing Visuals Aggregation Table
CREATE TABLE pairing_visuals (
    pairing_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    cigar_product_sku VARCHAR(64) NOT NULL,
    liquor_product_sku VARCHAR(64) NOT NULL,
    food_product_sku VARCHAR(64) DEFAULT NULL,
    generated_scene_url TEXT NOT NULL,
    fallback_generated_boolean BOOLEAN DEFAULT FALSE,
    composition_metadata JSONB DEFAULT '{}'::jsonb,
    last_rendered_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2.3 Generated Scenes Cache Ledger
CREATE TABLE generated_scenes (
    scene_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    prompt_payload TEXT NOT NULL,
    image_base64_payload TEXT DEFAULT NULL,
    cached_cdn_url TEXT NOT NULL,
    associated_atmosphere_id VARCHAR(64) NOT NULL,
    generation_metrics_json JSONB DEFAULT '{}'::jsonb,
    expires_at TIMESTAMP WITH TIME ZONE NOT NULL
);

-- 2.4 Inventory Image Mappings Record
CREATE TABLE inventory_images (
    mapping_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    product_sku VARCHAR(64) UNIQUE NOT NULL,
    primary_media_id UUID REFERENCES media_assets(asset_id) ON DELETE SET NULL,
    secondary_alpha_png_id UUID REFERENCES media_assets(asset_id) ON DELETE SET NULL,
    is_active_in_vault BOOLEAN DEFAULT TRUE,
    last_synchronized_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
