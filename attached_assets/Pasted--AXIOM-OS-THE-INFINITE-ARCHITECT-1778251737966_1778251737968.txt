import { EventEmitter } from "events";

class NeuralSubstrate extends EventEmitter {

  private static instance: NeuralSubstrate;

  constructor() {
    super();
  }

  static getInstance() {

    if (!NeuralSubstrate.instance) {
      NeuralSubstrate.instance =
        new NeuralSubstrate();
    }

    return NeuralSubstrate.instance;
  }

  emitSignal(
    signal: string,
    payload: any
  ) {
    this.emit(signal, payload);
  }

  listen(
    signal: string,
    callback: (payload: any) => void
  ) {
    this.on(signal, callback);
  }
}

export const neuralSubstrate =
  NeuralSubstrate.getInstance();

// server/services/VenueDNAService.ts

import { db } from "../db/client";
import {
  venueDNAProfiles,
} from "../db/schema";

export class VenueDNAService {

  static async evolveVenueDNA({
    venueId,
    signals,
  }: any) {

    let personality =
      "Sophisticated Lounge";

    if (signals.energy > 80) {
      personality =
        "High-Energy Social";
    }

    await db.insert(
      venueDNAProfiles
    ).values({
      venueId,
      personalityType: personality,
      atmosphereIntensity:
        signals.energy,
      pacingScore:
        signals.pacing,
      emotionalRhythm:
        signals.rhythm,
    });

    return personality;
  }
}
