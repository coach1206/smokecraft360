🔥 STEP 3 — ENGINE (THIS IS YOUR BRAIN)
/server/engine/recommendationEngine.js
import { scoreProducts } from "../services/scoringService.js";
import { applyBoosts } from "../services/boostService.js";
import { getPairings } from "./pairingEngine.js";
import { getFoodPairings } from "./foodEngine.js";
import { products } from "../data/fallbackData.js";

export async function runRecommendation(input) {
  const { category } = input;

  const filtered = products.filter(p => p.category === category);

  let scored = scoreProducts(filtered, input);

  scored = applyBoosts(scored);

  const top = scored.sort((a, b) => b.score - a.score).slice(0, 3);

  const pairings = getPairings(top[0]);
  const food = getFoodPairings(top[0]);

  return {
    recommendations: top,
    pairings,
    food
  };
}
🔥 STEP 4 — SCORING SERVICE (YOUR LOGIC)
/server/services/scoringService.js
export function scoreProducts(products, input) {
  return products.map(p => {
    let score = 0;

    // Flavor
    input.flavorPreferences.forEach(f => {
      if (p.flavorNotes.includes(f)) score += 3;
    });

    // Mood
    if (p.moodTags.includes(input.mood)) score += 4;

    // Strength
    const diff = Math.abs(p.strength - input.strength);
    if (diff === 0) score += 3;
    else if (diff === 1) score += 2;
    else if (diff === 2) score += 1;

    // Tier
    if (p.tier === "premium") score += 2;
    if (p.tier === "mid") score += 1;

    return { ...p, score };
  });
}
🔥 STEP 5 — BOOST SERVICE (MONETIZATION CORE)
/server/services/boostService.js
export function applyBoosts(products) {
  return products.map(p => {
    let bonus = p.boostLevel || 0;

    if (p.sponsored) bonus += 2;

    return {
      ...p,
      score: p.score + bonus
    };
  });
}
