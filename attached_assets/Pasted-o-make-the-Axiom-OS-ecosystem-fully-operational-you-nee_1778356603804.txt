3. The Sovereign Purge Logout (UserBadge.tsx)
This wires the user profile in the top-right corner to a secure system reset.

TypeScript
import React, { useState } from 'react';
import { useCraftContext } from '../context/CraftExperienceContext';

export const UserBadge = ({ userName }) => {
  const [showLogout, setShowLogout] = useState(false);
  const { purgeSession } = useCraftContext();

  const handleLogout = () => {
    // 1. Perform Sovereign Purge (Clear all flavor/mentor data)
    purgeSession();
    
    // 2. Redirect to Portal
    window.location.href = '/';
  };

  return (
    <div className="relative">
      <button 
        onClick={() => setShowLogout(!showLogout)}
        className="user-badge-premium"
      >
        {userName}
      </button>

      {showLogout && (
        <div className="logout-pill-floating" onClick={handleLogout}>
          <span className="text-gold">SOVEREIGN PURGE</span>
          <span className="text-xs text-silver">LOGOUT</span>
        </div>
      )}
    </div>
  );
};
