/* SESSION PERSISTENCE LOGIC */
const SovereignEngine = {
  saveSession: () => {
    localStorage.setItem('eeie_last_state', JSON.stringify({
      path: window.location.pathname,
      cart: currentCart,
      timestamp: Date.now()
    }));
  },
  restoreSession: (navigate) => {
    const saved = JSON.parse(localStorage.getItem('eeie_last_state'));
    if (saved && (Date.now() - saved.timestamp < 3600000)) {
      navigate(saved.path);
    }
  }
};

/* RIPPLE LOGOUT HANDLER */
const handleSovereignLogout = () => {
  SovereignEngine.saveSession();
  triggerRippleAnimation(() => {
    window.location.href = '/welcome-eeie';
  });
};
