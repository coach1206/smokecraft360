⚙️ BACKEND (EXPRESS ROUTES)
Admin: Assign package / features
// routes/admin-entitlements.ts
import express from "express";
import { db } from "../db";
import { venueEntitlements } from "../db/schema/venueEntitlements";
import { entitlementLogs } from "../db/schema/entitlementLogs";
import { v4 as uuid } from "uuid";

const router = express.Router();

router.post("/assign", async (req, res) => {
  const { venueId, packageId, features, pricing, adminId } = req.body;

  await db.insert(venueEntitlements).values({
    venueId,
    packageId,
    features,
    monthlyPrice: pricing.monthly,
    transactionFee: pricing.transaction,
    setupFee: pricing.setup,
  }).onConflictDoUpdate({
    target: venueEntitlements.venueId,
    set: {
      packageId,
      features,
      monthlyPrice: pricing.monthly,
      transactionFee: pricing.transaction,
      setupFee: pricing.setup,
    }
  });

  await db.insert(entitlementLogs).values({
    id: uuid(),
    venueId,
    adminId,
    action: "UPDATED_ENTITLEMENTS",
    data: { packageId, features, pricing },
  });

  res.json({ success: true });
});

export default router;
Middleware: Feature Check (CRITICAL)
// middleware/hasFeature.ts
import { db } from "../db";
import { venueEntitlements } from "../db/schema/venueEntitlements";

export const hasFeature = (featureId: string) => {
  return async (req, res, next) => {
    const venueId = req.user.venueId;

    const entitlement = await db.query.venueEntitlements.findFirst({
      where: (v) => v.venueId.eq(venueId),
    });

    if (!entitlement || !entitlement.features.includes(featureId)) {
      return res.status(403).json({ error: "Feature not enabled" });
    }

    next();
  };
};

🎨 FRONTEND (REACT ADMIN PANEL)
Feature Selector UI
// AdminEntitlements.tsx
const FEATURES = [
  "SMOKECRAFT",
  "POURCRAFT",
  "AI_PAIRING",
  "DYNAMIC_PRICING",
  "INVENTORY",
  "ANALYTICS",
  "LOYALTY",
  "CAMPAIGNS"
];

export default function AdminEntitlements() {
  const [selected, setSelected] = useState<string[]>([]);
  const [pricing, setPricing] = useState({
    monthly: 0,
    transaction: 0,
    setup: 0
  });

  const toggle = (feature: string) => {
    setSelected(prev =>
      prev.includes(feature)
        ? prev.filter(f => f !== feature)
        : [...prev, feature]
    );
  };

  return (
    <div>
      <h2>Assign Features</h2>

      {FEATURES.map(f => (
        <label key={f}>
          <input
            type="checkbox"
            checked={selected.includes(f)}
            onChange={() => toggle(f)}
          />
          {f}
        </label>
      ))}

      <h3>Pricing</h3>
      <input placeholder="Monthly"
        onChange={e => setPricing({...pricing, monthly: +e.target.value})} />
      <input placeholder="Transaction %"
        onChange={e => setPricing({...pricing, transaction: +e.target.value})} />
      <input placeholder="Setup Fee"
        onChange={e => setPricing({...pricing, setup: +e.target.value})} />

      <button onClick={async () => {
        await fetch("/api/admin/assign", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            venueId: "venue_123",
            features: selected,
            pricing
          })
        });
      }}>
        Save
      </button>
    </div>
  );
}
