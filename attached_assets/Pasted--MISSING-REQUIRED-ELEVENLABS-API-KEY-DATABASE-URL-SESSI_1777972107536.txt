export type BuildInput = {  strength:number; flavorMatch:number; balance:number; finish:number;  timeTaken:number; converted:boolean;};export function scoreBuild(i:BuildInput){  const quality=(i.strength+i.flavorMatch)/2;  const consistency=(i.balance+i.finish)/2;  const speed=i.timeTaken<600?5:i.timeTaken<1200?3:1;  const conv=i.converted?5:2;  const score = quality*.4 + i.balance*.2 + consistency*.2 + speed*.1 + conv*.1;  let feedback="";  if(score<2.5) feedback="That blend fails. Fix it.";  else if(score<3.5) feedback="Close. Adjust pairing.";  else if(score<4.5) feedback="Strong. Refine finish.";  else feedback="Elite build.";  return { score:+score.toFixed(2), feedback };}
