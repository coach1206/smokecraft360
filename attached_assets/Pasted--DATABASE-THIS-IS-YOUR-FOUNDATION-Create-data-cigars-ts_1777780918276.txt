🔥 2. MATCHING ENGINE (THIS IS THE BRAIN)

Create:
/utils/matchEngine.ts

export function scoreItem(userPrefs: any, item: any) {
  let score = 0;

  // Flavor match
  const matches = item.flavors.filter((f: string) =>
    userPrefs.flavors.includes(f)
  );
  score += matches.length * 20;

  // Strength match
  if (userPrefs.strength === item.strength) {
    score += 25;
  }

  // Mood influence
  if (userPrefs.mood === "relaxed" && item.strength === "mild") {
    score += 10;
  }

  if (userPrefs.mood === "bold" && item.strength === "full") {
    score += 10;
  }

  return score;
}

export function getBestMatch(userPrefs: any, list: any[]) {
  let best = null;
  let highest = 0;

  for (const item of list) {
    const score = scoreItem(userPrefs, item);

    if (score > highest) {
      highest = score;
      best = item;
    }
  }

  return best;
}
🔥 3. CONNECT TO YOUR UI

In your Reveal Screen:

import { cigars } from "../data/cigars";
import { drinks } from "../data/drinks";
import { getBestMatch } from "../utils/matchEngine";
import { useExperience } from "../context/ExperienceContext";

export default function Reveal() {
  const { selections } = useExperience();

  const userPrefs = {
    flavors: selections.flavor || [],
    strength: selections.strength,
    mood: selections.mood
  };

  const cigar = getBestMatch(userPrefs, cigars);
  const drink = getBestMatch(userPrefs, drinks);

  return (
    <div className="reveal">
      <img src={cigar.image} />
      <h1>{cigar.name}</h1>

      <img src={drink.image} />
      <h2>{drink.name}</h2>
    </div>
  );
}
