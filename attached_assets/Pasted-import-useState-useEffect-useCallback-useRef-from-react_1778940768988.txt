import { useState, useEffect, useCallback, useRef } from 'react';

interface BiometricStream {
  heartRate: number;
  oxygenSaturation: number;
  stressIndex: number;
  coreTemperature: number;
}

export const useSovereignBiometrics = (venueId: string) => {
  const [vitals, setVitals] = useState<BiometricStream | null>(null);
  const [isActive, setIsActive] = useState(false);
  const [lastDispatch, setLastDispatch] = useState<string | null>(null);
  
  const timerRef = useRef<NodeJS.Timeout | null>(null);
  const debounceGuardRef = useRef<boolean>(false);

  // Generates 10Hz stream data via sine-wave modeling + Gaussian noise
  const generateMockVitals = (tick: number): BiometricStream => {
    const hrBase = 70 + Math.sin(tick * 0.05) * 20;
    const gaussNoise = (Math.random() - 0.5) * 4;
    
    return {
      heartRate: Math.round(Math.max(45, Math.min(180, hrBase + gaussNoise))),
      oxygenSaturation: Math.round(Math.max(88, Math.min(100, 98 + Math.sin(tick * 0.01) * 2))),
      stressIndex: Math.round(Math.max(0, Math.min(100, 45 + Math.cos(tick * 0.03) * 25 + (Math.random() * 5)))),
      coreTemperature: parseFloat((37.0 + Math.sin(tick * 0.005) * 0.8).toFixed(1))
    };
  };

  const dispatchAmbientOverride = useCallback(async () => {
    if (debounceGuardRef.current) return;

    debounceGuardRef.current = true;
    const now = new Date().toISOString();
    setLastDispatch(now);

    try {
      await fetch(`/api/intelligence/ambient/${venueId}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ sceneId: 'low_amber_recovery' })
      });
    } catch (err) {
      console.error('SARV ambient override dispatch failure:', err);
    }

    // Enforce strict 8-second debounce to prevent endpoint saturation
    setTimeout(() => {
      debounceGuardRef.current = false;
    }, 8000);
  }, [venueId]);

  const start = useCallback(() => {
    if (isActive) return;
    setIsActive(true);
    let tick = 0;

    // 100ms interval matching the 10Hz hardware tactility floor
    timerRef.current = setInterval(() => {
      const currentVitals = generateMockVitals(tick++);
      setVitals(currentVitals);

      // Evaluate logic thresholds for immediate physical adaptation trigger
      if (currentVitals.stressIndex > 60 || currentVitals.heartRate > 95) {
        dispatchAmbientOverride();
      }
    }, 1000); 
  }, [isActive, dispatchAmbientOverride]);

  const stop = useCallback(() => {
    if (timerRef.current) clearInterval(timerRef.current);
    setIsActive(false);
  }, []);

  useEffect(() => {
    return () => { if (timerRef.current) clearInterval(timerRef.current); };
  }, []);

  return { vitals, isActive, lastDispatch, start, stop };
};
