// server/services/fakeDataGenerator.ts
import { logEvent } from "../utils/logEvent";

const sampleFlavors = ["earthy", "smooth", "spicy"];
const sampleStrength = ["light", "medium", "full"];

export function startFakeData(venueId: string) {
  setInterval(() => {
    const sessionId = "demo_" + Math.random();

    logEvent({
      venueId,
      sessionId,
      eventType: "SESSION_START",
    });

    logEvent({
      venueId,
      sessionId,
      eventType: "PRODUCT_VIEWED",
      metadata: {
        flavor: sampleFlavors[Math.floor(Math.random() * 3)],
        strength: sampleStrength[Math.floor(Math.random() * 3)],
      },
    });

    logEvent({
      venueId,
      sessionId,
      eventType: "PRODUCT_PURCHASED",
      price: Math.random() * 30,
    });
  }, 5000);
}
