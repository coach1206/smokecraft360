🎤 AUDIO SYSTEM (SINGLE INSTANCE ONLY)
const audioRef = useRef(null);

function speak(text) {
  fetch("/api/voice", {
    method: "POST",
    body: JSON.stringify({ text })
  })
    .then(res => res.json())
    .then(data => {
      if (audioRef.current) {
        audioRef.current.pause();
      }

      const audio = new Audio(data.url);
      audioRef.current = audio;
      audio.play();
    });
}

  if (context.mood === "premium") base += "?gold";

}

💰 PRICING ENGINE
function dynamicPrice(base, data) {
  let adj = 0;

  if (data.demand) adj += 0.08;
  if (data.lowSales) adj -= 0.05;

  return base * (1 + adj);
}

function profit(item) {

    margin: (item.price - item.cost) / item.price
