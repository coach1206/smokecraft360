🍺 BREWCRAFT PAGE (MAIN FLOW)

pages/BrewCraftPage.tsx

import React, { useState } from "react";
import BrewCraftCards from "../components/BrewCraftCards";
import PourCraftUpsell from "../components/PourCraftUpsell";

export default function BrewCraftPage() {
  const [result, setResult] = useState<any>(null);

  function handleSelect(type: string) {
    fetch(`/api/pairing/${type}`)
      .then((res) => res.json())
      .then(setResult);

    fetch("/api/brewcraft/selection", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ type }),
    });
  }

  return (
    <div className="p-6 text-white">
      <h1 className="text-2xl mb-4">BrewCraft Experience</h1>

      <BrewCraftCards onSelect={handleSelect} />

      {result && (
        <div className="mt-6 bg-zinc-900 p-4 rounded-xl">
          <h2 className="text-xl">{result.name}</h2>
          <p>{result.beer} + {result.cigar}</p>
          <p className="opacity-70">{result.vibe}</p>

          <PourCraftUpsell base={result} />
        </div>
      )}
    </div>
  );
}
