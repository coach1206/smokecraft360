// server/db/schema.ts

import {
  pgTable,
  serial,
  text,
  integer,
  timestamp,
  boolean,
  json
} from "drizzle-orm/pg-core";

export const devices = pgTable("devices", {
  id: serial("id").primaryKey(),
  name: text("name"),
  battery: integer("battery"),
  status: text("status"),
  location: text("location"),
  active: boolean("active").default(true)
});

export const pairings = pgTable("pairings", {
  id: serial("id").primaryKey(),
  cigarName: text("cigar_name"),
  liquorName: text("liquor_name"),
  loungeImage: text("lounge_image"),
  cigarImage: text("cigar_image"),
  liquorImage: text("liquor_image"),
  notes: text("notes"),
  strength: integer("strength")
});

export const environmentProfiles = pgTable("environment_profiles", {
  id: serial("id").primaryKey(),
  name: text("name"),
  lighting: integer("lighting"),
  music: text("music"),
  scent: text("scent"),
  temperature: integer("temperature"),
  humidity: integer("humidity")
});

export const activityFeed = pgTable("activity_feed", {
  id: serial("id").primaryKey(),
  message: text("message"),
  image: text("image"),
  createdAt: timestamp("created_at").defaultNow()
});

// server/index.ts

import express from "express";
import http from "http";
import cors from "cors";
import { Server } from "socket.io";

import dashboardRoutes from "./routes/dashboard";
import pairingRoutes from "./routes/pairings";
import deviceRoutes from "./routes/devices";
import environmentRoutes from "./routes/environment";

const app = express();

app.use(cors());
app.use(express.json());

app.use("/api/dashboard", dashboardRoutes);
app.use("/api/pairings", pairingRoutes);
app.use("/api/devices", deviceRoutes);
app.use("/api/environment", environmentRoutes);

const server = http.createServer(app);

const io = new Server(server, {
  cors: {
    origin: "*"
  }
});

io.on("connection", (socket) => {
  console.log("Touchscreen connected");

  socket.emit("system-status", {
    live: true
  });

  setInterval(() => {
    socket.emit("ambient-update", {
      lighting: Math.floor(Math.random() * 100),
      humidity: 52,
      temp: 70
    });
  }, 5000);
});

server.listen(5000, () => {
  console.log("E.A.T System running on port 5000");
});

import { Router } from "express";
import multer from "multer";

const router = Router();

const storage = multer.diskStorage({
  destination: "./uploads",
  filename: (_, file, cb) => {
    cb(null, Date.now() + "-" + file.originalname);
  }
});

const upload = multer({ storage });

router.post("/upload", upload.single("image"), (req, res) => {
  res.json({
    success: true,
    file: req.file
  });
});

export default router;
