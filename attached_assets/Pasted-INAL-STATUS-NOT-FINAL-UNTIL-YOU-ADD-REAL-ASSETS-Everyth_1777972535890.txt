📁 BACKEND (Node + Express)
server/index.ts
import express from "express";
import bodyParser from "body-parser";
import cors from "cors";
import fs from "fs";

const app = express();
app.use(cors());
app.use(bodyParser.json());

const DB_FILE = "./db.json";

// INIT DB
if (!fs.existsSync(DB_FILE)) {
  fs.writeFileSync(DB_FILE, JSON.stringify({ builds: [] }));
}

// SAVE BUILD
app.post("/api/build", (req, res) => {
  const data = JSON.parse(fs.readFileSync(DB_FILE, "utf-8"));
  const newBuild = {
    id: Date.now(),
    ...req.body,
    createdAt: new Date()
  };
  data.builds.push(newBuild);
  fs.writeFileSync(DB_FILE, JSON.stringify(data, null, 2));
  res.json(newBuild);
});

// GET BUILDS
app.get("/api/builds", (req, res) => {
  const data = JSON.parse(fs.readFileSync(DB_FILE, "utf-8"));
  res.json(data.builds);
});

app.listen(5000, () => console.log("Server running"));

CraftVisualizer.tsx
import { useState } from "react";
import axios from "axios";

export default function CraftVisualizer() {
  const [wood, setWood] = useState("cedar");
  const [trim, setTrim] = useState("gold");
  const [interior, setInterior] = useState("cream");
  const [brand, setBrand] = useState("MY BRAND");
  const [reveal, setReveal] = useState(false);

  const saveBuild = async () => {
    await axios.post("http://localhost:5000/api/build", {
      wood, trim, interior, brand
    });
  };

  return (
    <div className="bg-black h-screen flex flex-col items-center justify-center text-white">

      {/* VISUAL STACK */}
      <div className={`relative w-[500px] h-[300px] ${reveal ? "scale-110 transition-all duration-700" : ""}`}>

        <img src="/assets/smokecraft/box_open_base.png" className="absolute w-full" />
        <img src={`/assets/smokecraft/wood/${wood}.png`} className="absolute w-full mix-blend-multiply" />
        <img src={`/assets/smokecraft/overlays/${trim}.png`} className="absolute w-full" />
        <img src={`/assets/smokecraft/interior/${interior}.png`} className="absolute w-full" />
        <img src="/assets/smokecraft/cigars/cigars_bundle.png" className="absolute w-full" />

        <div className="absolute bottom-6 left-6 text-yellow-400 text-xl">
          {brand}
        </div>

        {reveal && (
          <div className="absolute inset-0 bg-yellow-500 opacity-20 blur-2xl animate-pulse" />
        )}

      </div>

      {/* CONTROLS */}
      <div className="mt-6 space-y-3">

        <input
          value={brand}
          onChange={(e) => setBrand(e.target.value)}
          className="px-3 py-2 text-black"
        />

        <div className="flex gap-2">
          {["cedar","mahogany","walnut","ebony"].map(w =>
            <button key={w} onClick={() => setWood(w)} className="bg-gray-800 px-3 py-2">
              {w}
            </button>
          )}
        </div>

        <div className="flex gap-2">
          {["gold","platinum"].map(t =>
            <button key={t} onClick={() => setTrim(t)} className="bg-gray-800 px-3 py-2">
              {t}
            </button>
          )}
        </div>

        <div className="flex gap-2">
          {["cream","velvet","linen"].map(i =>
            <button key={i} onClick={() => setInterior(i)} className="bg-gray-800 px-3 py-2">
              {i}
            </button>
          )}
        </div>

        <button
          onClick={() => {
            setReveal(true);
            saveBuild();
          }}
          className="bg-yellow-500 text-black px-6 py-3"
        >
          Reveal My Brand
        </button>

      </div>

    </div>
  );
}
