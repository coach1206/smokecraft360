## STEP 2 — EXPERIENCE DECISION ENGINE

Create:
server/services/smokeCraftDecisionEngine.ts

Add:

import { AI_BEHAVIOR_PROFILE } from "../config/aiBehavior";

export function validateExperience(experience: any) {
if (!experience) return false;
if (!experience.cigar || !experience.drink) return false;
return true;
}

export function validatePairing(pairing: any) {
if (!pairing) return false;

// simple flavor compatibility logic (expand later)
if (pairing.strengthMismatch === true) return false;

return true;
}

export function validateUpsell(user: any, product: any) {
if (!user || !product) return false;

if (user.level === "beginner" && product.price > 150) return false;

return true;
}

export function validateInventory(item: any) {
if (!item) return false;
if (item.available === false) return false;
if (item.quantity !== undefined && item.quantity <= 0) return false;

return true;
}

export function buildBehaviorContext(context: string) {
return {
identity: AI_BEHAVIOR_PROFILE.identity,
principles: AI_BEHAVIOR_PROFILE.principles,
context
};
}
